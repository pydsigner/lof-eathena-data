function|script|DisplayMOTD
{
    if($@MOTD_Disabled || $@MOTD$[0] == "") goto L_Return;
    goto L_MOTD;

L_MOTD:
    message strcharinfo(0), "##7Server : " + $@MOTD$[@motd_index];
    set @motd_index, @motd_index + 1;
    if($@MOTD$[@motd_index] == "") goto L_Return;
    goto L_MOTD;

L_Return:
    return;
}
